🔧 Google Apps Script Setup

This app saves all attendance and employee data directly to your Google Sheet using the URL you configured. Follow the steps below to set up the Apps Script backend.

1
Open Google Sheets — create a new spreadsheet named AttendIQ. It needs two sheets: Employees and Attendance.
2
Open Apps Script — click Extensions → Apps Script. Delete all existing code and paste the script below.
3
Deploy as Web App — click Deploy → New Deployment → Web App → Execute as Me → Who has access: Anyone → Deploy.
4
Done! — Your Web App URL is already configured in this app:
https://script.google.com/macros/s/AKfycbzntySVZfGRy5PH8CHVvyksC6iBO6V3UDaCa8Gufu9EGvW0BgQx2RVhicpq-FiW8xdtMw/exec
📋 Apps Script Code — paste this in Extensions → Apps Script:
const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const p = e.parameter || {}; const action = p.action || (e.postData ? JSON.parse(e.postData.contents).action : ''); const data = e.postData ? JSON.parse(e.postData.contents) : p; try { let result; if (action === 'ping') result = { status: 'ok', message: 'AttendIQ connected' }; else if (action === 'getAll') result = getAllData(); else if (action === 'addEmp') result = addEmployee(data); else if (action === 'logAttend') result = logAttendance(data); else if (action === 'deleteEmp') result = deleteEmployee(data); else result = { status: 'error', message: 'Unknown action: ' + action }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status:'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getAllData() { const ss = SHEET_ID; const empSheet = ss.getSheetByName('Employees') || ss.insertSheet('Employees'); const attSheet = ss.getSheetByName('Attendance') || ss.insertSheet('Attendance'); // Setup headers if empty if (empSheet.getLastRow() === 0) { empSheet.appendRow(['ID','Name','Email','Department','Role','Color','CreatedAt']); } if (attSheet.getLastRow() === 0) { attSheet.appendRow(['ID','EmpID','Name','Department','Type','Date','Time','Timestamp','Duration','City','Latitude','Longitude','Method']); } const employees = dataToObjects(empSheet); const attendance = dataToObjects(attSheet); return { status:'ok', employees, attendance }; } function addEmployee(data) { const ss = SHEET_ID; const sheet = ss.getSheetByName('Employees') || ss.insertSheet('Employees'); if (sheet.getLastRow() === 0) sheet.appendRow(['ID','Name','Email','Department','Role','Color','CreatedAt']); // Check duplicate email const existing = dataToObjects(sheet); if (existing.find(e => e.Email && e.Email.toLowerCase() === data.email.toLowerCase())) return { status:'error', message:'Email already exists' }; sheet.appendRow([data.id, data.name, data.email, data.dept, data.role, data.color, data.createdAt]); return { status:'ok', message:'Employee added' }; } function logAttendance(data) { const ss = SHEET_ID; const sheet = ss.getSheetByName('Attendance') || ss.insertSheet('Attendance'); if (sheet.getLastRow() === 0) sheet.appendRow(['ID','EmpID','Name','Department','Type','Date','Time','Timestamp','Duration','City','Latitude','Longitude','Method']); sheet.appendRow([ data.id, data.empId, data.name, data.dept, data.type.toUpperCase(), data.date, data.time, data.timestamp, data.duration || '', data.city || '', data.lat || '', data.lng || '', data.method || 'face' ]); return { status:'ok', message:'Attendance logged' }; } function deleteEmployee(data) { const ss = SHEET_ID; const sheet = ss.getSheetByName('Employees'); if (!sheet) return { status:'error', message:'Sheet not found' }; const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.id) { sheet.deleteRow(i + 1); break; } } return { status:'ok', message:'Employee deleted' }; } function dataToObjects(sheet) { if (sheet.getLastRow() < 2) return []; const [headers, ...rows] = sheet.getDataRange().getValues(); return rows.map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); }
